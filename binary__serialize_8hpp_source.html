272// lower-level function template that performs the actual buffer manipulation and
273// marshalling of a single value, with an ADL tag for full namespace inclusion in the
274// overload set; this function template is not called directly by application code, and
275// is only used with arithmetic values or a std::byte
276template <typename CastValType, typename T, typename Buf = chops::mutable_shared_buffer>
277Buf& marshall(Buf& buf, const T& val, adl_tag) {
278 auto old_sz = buf.size();
279 buf.resize(old_sz + sizeof(CastValType));
280 append_val(buf.data()+old_sz, static_cast<CastValType>(val));
281 return buf;
282}

339Buf& marshall(Buf& buf, const std::optional<T>& val) {
340 marshall<CastBoolType>(buf, val.has_value());

369// efficiently append a buffer of bytes to the end of the existing buffer

372 auto old_sz = buf.size();
